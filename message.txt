CREATE TABLE customers
(
    customer_id   SERIAL                                NOT NULL,
    customer_type VARCHAR(20)                           NOT NULL,
    first_name    VARCHAR(100),
    last_name     VARCHAR(100),
    company_name  VARCHAR(200),
    email         VARCHAR(255) UNIQUE,
    phone         VARCHAR(30),
    date_of_birth DATE,
    status        VARCHAR(20) DEFAULT 'active'          NOT NULL,
    created_at    TIMESTAMP   DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at    TIMESTAMP   DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (customer_id),
    CONSTRAINT chk_customers_type_value CHECK (customer_type IN ('individual', 'business')),
    CONSTRAINT chk_customers_type_fields CHECK (
        (customer_type = 'individual' AND first_name IS NOT NULL AND last_name IS NOT NULL)
        OR
        (customer_type = 'business' AND company_name IS NOT NULL)
    )
);

CREATE TABLE customer_addresses
(
    address_id   SERIAL                              NOT NULL,
    customer_id  INT                                 NOT NULL,
    address_type VARCHAR(20)                         NOT NULL,
    country      VARCHAR(100)                        NOT NULL,
    city         VARCHAR(100)                        NOT NULL,
    street       VARCHAR(255)                        NOT NULL,
    postal_code  VARCHAR(20),
    latitude     NUMERIC(9, 6),
    longitude    NUMERIC(9, 6),
    is_primary   BOOLEAN   DEFAULT FALSE             NOT NULL,
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (address_id),
    CONSTRAINT chk_customer_addresses_type CHECK (address_type IN ('billing', 'service', 'home', 'office', 'other')),
    CONSTRAINT fk_customer_addresses_customer FOREIGN KEY (customer_id) REFERENCES customers (customer_id)
);

CREATE TABLE billing_cycles
(
    billing_cycle_id SERIAL                              NOT NULL,
    cycle_name       VARCHAR(50)                         NOT NULL UNIQUE,
    day_of_month     INT                                 NOT NULL,
    grace_days       INT       DEFAULT 0                 NOT NULL,
    is_active        BOOLEAN   DEFAULT TRUE              NOT NULL,
    created_at       TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (billing_cycle_id),
    CONSTRAINT chk_billing_cycles_day CHECK (day_of_month BETWEEN 1 AND 31),
    CONSTRAINT chk_billing_cycles_grace CHECK (grace_days >= 0)
);

CREATE TABLE accounts
(
    account_id       SERIAL                                   NOT NULL,
    customer_id      INT                                      NOT NULL,
    account_number   VARCHAR(50)                              NOT NULL UNIQUE,
    account_status   VARCHAR(20)    DEFAULT 'active'          NOT NULL,
    credit_limit     NUMERIC(12, 2) DEFAULT 0.00              NOT NULL,
    current_balance  NUMERIC(12, 2) DEFAULT 0.00              NOT NULL,
    billing_cycle_id INT,
    created_at       TIMESTAMP      DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at       TIMESTAMP      DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (account_id),
    CONSTRAINT fk_accounts_customer FOREIGN KEY (customer_id) REFERENCES customers (customer_id),
    CONSTRAINT fk_accounts_billing_cycle FOREIGN KEY (billing_cycle_id) REFERENCES billing_cycles (billing_cycle_id)
);

    subscription_id      INT,

    ticket_type          VARCHAR(50)                           NOT NULL,
    subject              VARCHAR(255)                          NOT NULL,
    description          TEXT,
    priority             VARCHAR(20) DEFAULT 'medium'          NOT NULL,
    status               VARCHAR(20) DEFAULT 'open'            NOT NULL,
    created_at           TIMESTAMP   DEFAULT CURRENT_TIMESTAMP NOT NULL,
    closed_at            TIMESTAMP,
    parent_ticket_id     INT,
    PRIMARY KEY (ticket_id),
    CONSTRAINT chk_crm_tickets_times CHECK (closed_at IS NULL OR closed_at >= created_at),
    CONSTRAINT fk_crm_tickets_customer FOREIGN KEY (customer_id) REFERENCES customers (customer_id),
    CONSTRAINT fk_crm_tickets_account FOREIGN KEY (account_id) REFERENCES accounts (account_id),
    CONSTRAINT fk_crm_tickets_subscription FOREIGN KEY (subscription_id) REFERENCES subscriptions (subscription_id),
    CONSTRAINT fk_crm_tickets_employee FOREIGN KEY (assigned_employee_id) REFERENCES employees (employee_id),
    CONSTRAINT fk_crm_tickets_parent FOREIGN KEY (parent_ticket_id) REFERENCES crm_tickets (ticket_id)
);

CREATE TABLE crm_interactions
(
    interaction_id   SERIAL                              NOT NULL,
    ticket_id        INT                                 NOT NULL,
    employee_id      INT,
    interaction_type VARCHAR(50)                         NOT NULL,
    channel          VARCHAR(30)                         NOT NULL,
    interaction_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    notes            TEXT,
    old_status       VARCHAR(20),
    new_status       VARCHAR(20),
    PRIMARY KEY (interaction_id),
    CONSTRAINT fk_crm_interactions_ticket FOREIGN KEY (ticket_id) REFERENCES crm_tickets (ticket_id),
    CONSTRAINT fk_crm_interactions_employee FOREIGN KEY (employee_id) REFERENCES employees (employee_id)
);

CREATE TABLE ticket_status_history
(
    ticket_status_history_id SERIAL                              NOT NULL,
    ticket_id                INT                                 NOT NULL,
    old_status               VARCHAR(20),
    new_status               VARCHAR(20)                         NOT NULL,
    changed_at               TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    changed_by_employee_id   INT,
    comment                  TEXT,
    PRIMARY KEY (ticket_status_history_id),
    CONSTRAINT fk_ticket_history_ticket FOREIGN KEY (ticket_id) REFERENCES crm_tickets (ticket_id),

);

CREATE TABLE employee_assignments
(
    assignment_id   SERIAL                         NOT NULL,
    employee_id     INT                            NOT NULL,
    ticket_id       INT,
    outage_id       INT,
    assignment_type VARCHAR(50)                    NOT NULL,
    start_time      TIMESTAMP                      NOT NULL,
    end_time        TIMESTAMP,
    status          VARCHAR(20) DEFAULT 'assigned' NOT NULL,
    PRIMARY KEY (assignment_id),
    CONSTRAINT chk_employee_assignments_times CHECK (end_time IS NULL OR end_time >= start_time),
    CONSTRAINT fk_employee_assignments_employee FOREIGN KEY (employee_id) REFERENCES employees (employee_id),
    CONSTRAINT fk_employee_assignments_ticket FOREIGN KEY (ticket_id) REFERENCES crm_tickets (ticket_id),
    CONSTRAINT fk_employee_assignments_outage FOREIGN KEY (outage_id) REFERENCES outages (outage_id)
