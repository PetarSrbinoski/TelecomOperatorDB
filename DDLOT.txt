CREATE TABLE customers
(
    customer_id   BIGSERIAL                                NOT NULL,
    customer_type TEXT                           NOT NULL,
    first_name    TEXT,
    last_name     TEXT,
    company_name  TEXT,
    email         TEXT UNIQUE,
    phone         TEXT,
    date_of_birth DATE,
    status        TEXT DEFAULT 'active'          NOT NULL,
    created_at    TIMESTAMP   DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at    TIMESTAMP   DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (customer_id),
    CONSTRAINT chk_customers_type_value CHECK (customer_type IN ('individual', 'business')),
    CONSTRAINT chk_customers_type_fields CHECK (
        (customer_type = 'individual' AND first_name IS NOT NULL AND last_name IS NOT NULL)
            OR
        (customer_type = 'business' AND company_name IS NOT NULL)
        )
);

CREATE TABLE customer_addresses
(
    address_id   BIGSERIAL                              NOT NULL,
    customer_id  BIGINT                                 NOT NULL,
    address_type TEXT                         NOT NULL,
    country      TEXT                        NOT NULL,
    city         TEXT                        NOT NULL,
    street       TEXT                        NOT NULL,
    postal_code  TEXT,
    latitude     NUMERIC,
    longitude    NUMERIC,
    is_primary   BOOLEAN   DEFAULT FALSE             NOT NULL,
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (address_id),
    CONSTRAINT chk_customer_addresses_type CHECK (address_type IN ('billing', 'service', 'home', 'office', 'other')),
    CONSTRAINT fk_customer_addresses_customer FOREIGN KEY (customer_id) REFERENCES customers (customer_id)
);

CREATE TABLE billing_cycles
(
    billing_cycle_id BIGSERIAL                              NOT NULL,
    cycle_name       TEXT                         NOT NULL UNIQUE,
    day_of_month     INT                                 NOT NULL,
    grace_days       INT       DEFAULT 0                 NOT NULL,
    is_active        BOOLEAN   DEFAULT TRUE              NOT NULL,
    created_at       TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    PRIMARY KEY (billing_cycle_id),
    CONSTRAINT chk_billing_cycles_day CHECK (day_of_month BETWEEN 1 AND 31),
    CONSTRAINT chk_billing_cycles_grace CHECK (grace_days >= 0)
);

CREATE TABLE accounts
(
    account_id       BIGSERIAL                                   NOT NULL,

    CONSTRAINT fk_accounts_billing_cycle FOREIGN KEY (billing_cycle_id) REFERENCES billing_cycles (billing_cycle_id)
);

    interaction_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    notes            TEXT,
    old_status       TEXT,
    new_status       TEXT,
    PRIMARY KEY (interaction_id),
    CONSTRAINT fk_crm_interactions_ticket FOREIGN KEY (ticket_id) REFERENCES crm_tickets (ticket_id),
    CONSTRAINT fk_crm_interactions_employee FOREIGN KEY (employee_id) REFERENCES employees (employee_id)
);

CREATE TABLE ticket_status_history
(
    ticket_status_history_id BIGSERIAL                              NOT NULL,
    ticket_id                BIGINT                                 NOT NULL,
    old_status               TEXT,
    new_status               TEXT                         NOT NULL,
    changed_at               TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    changed_by_employee_id   BIGINT,
    comment                  TEXT,
    PRIMARY KEY (ticket_status_history_id),
    CONSTRAINT fk_ticket_history_ticket FOREIGN KEY (ticket_id) REFERENCES crm_tickets (ticket_id),
    CONSTRAINT fk_ticket_history_employee FOREIGN KEY (changed_by_employee_id) REFERENCES employees (employee_id)
);

CREATE TABLE employee_assignments
(
    assignment_id   BIGSERIAL                         NOT NULL,
    employee_id     BIGINT                            NOT NULL,
    ticket_id       BIGINT,
    outage_id       BIGINT,
    assignment_type TEXT                    NOT NULL,
    start_time      TIMESTAMP                      NOT NULL,
    end_time        TIMESTAMP,
    status          TEXT DEFAULT 'assigned' NOT NULL,
    PRIMARY KEY (assignment_id),
    CONSTRAINT chk_employee_assignments_times CHECK (end_time IS NULL OR end_time >= start_time),
    CONSTRAINT fk_employee_assignments_employee FOREIGN KEY (employee_id) REFERENCES employees (employee_id),
    CONSTRAINT fk_employee_assignments_ticket FOREIGN KEY (ticket_id) REFERENCES crm_tickets (ticket_id),
    CONSTRAINT fk_employee_assignments_outage FOREIGN KEY (outage_id) REFERENCES outages (outage_id)
);
